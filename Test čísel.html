from pyscript import document # import objektu document jako zástupce objektu dokument webovských stránek (built-in helper) def testuj(event): # fce testuj udalost vstup = document.querySelector("#retezec") # do vstupu dej první Element v dokumentu, který odpovídá zadanému selektoru retezec c = vstup.value # hodnotu retezce dej do c vystup = document.querySelector("#odezva") # do vystupu dej první Element v dokumentu, který odpovídá zadanému selektoru odezva vystup.innerText = analyza(c) # do vlastnosti innerText objektu vystup dej hodnotu vrácenou funkcí analyza(c) def analyza(zadani): # fce analyza zadaného řetězce try: # blok testu kódu, zda obsahuje nějaké chyby a = float(zadani) # prevod zadaneho retezce na realne cislo except ValueError: # vyjimka - nesprávná hodnota v zadaném datovém typu b = "→ to není raálné číslo !!!" # hlaska, pokud neni retezec realne cislo return b # vrat tuto hlasku b if a > 0: # analyza realneho cisla ..... b = ("→ číslo je kladné") elif a < 0: b = ("→ číslo je záporné") else: b = ("→ toto je nula") return b # vrat tuto hlasku b
